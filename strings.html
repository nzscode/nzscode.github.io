# Splitting Strings into parts

print(part1)
print(part2)

# String to Dicts

string = "Mary had a little lamb"

split_string = string.split(" ")

string_dict = {}

for i in range(len(split_string)):
    new_key = "word " + str(i)
    string_dict[new_key] = split_string[i]

print(string_dict)


    Output:

    {'word 0': 'Mary', 'word 1': 'had', 'word 2': 'a', 'word 3': 'little', 'word 4': 'lamb'} 


                        
# String to CSV

import csv

split_string = string.split(" ")

fields = ["id", "words"]

path = "../Resources/"
with open(path + "string.csv", 'w') as file:

for i in range(len(string_split)):
    string_data.append((i, string_split[i]))

path = "../Resources/"

pdf_entries = []
headers = ("id", "word")
