# TODO: 1. We have to import the Modules and Packages we need.
from reportlab.platypus import SimpleDocTemplate, TableStyle
from reportlab.platypus import Paragraph, Spacer, Table, Image
from reportlab.lib.styles import getSampleStyleSheet
from reportlab.lib import colors

names = [('Sophia', 'Smith'), ('Liam', 'Kim'), ('Olivia', 'Garcia'), ('Noah', 'Muller')]
jobs = [("Teacher", "Da Vinci PS"), ("Student", "Dunbarton High"), ("Nurse", "Rouge Health"),
        ("Zoo keeper", "Toronto Zoo")]

# TODO: 2. We must give our new pdf a name and point the path to the folder where we want to store our new pdf.
path = "../Resources/"
pdf = SimpleDocTemplate(path + "tuples.pdf")

# TODO: 3. We must assign a variable for the Style Sheet Template we will use.
styles = getSampleStyleSheet()

# TODO: 4. We can give our pdf a title, and a style.
pdf_title = Paragraph("Full Names", styles["h1"])

# TODO: 5. We have to make a list where the data will be sorted and stored.
people = []

# TODO: 6. We need to assign table headers to a list and append it to the data list.
headers = ("First Name", "Last Name", "Job", "Workplace")
people.append(headers)
for i in range(len(names)):
    person = (names[i][0], names[i][1], jobs[i][0], jobs[i][1])
    people.append(person)

# TODO: 7. We need to assign a variable where we will declare the data that is to be the table.
                #We will also specifiy a corner radius(top-left, top-right, bottom-left, bottom-right)
pdf_table = Table(data=people, cornerRadii=(20, 3, 3, 33))

# TODO: 8. We assign a variable to hold all our new style info.
header_style = TableStyle([
    # cells (0,0) to (3,0) are aligned center
    ('ALIGN', (0, 0), (3, 0), 'CENTER'),

    # cells (0,0) to (3,0) have a background color of (r,g,b,a(btwn 0-1)),
        #rgb must be in decimals so rgb values must each be divided by 255
    ('BACKGROUND', (0, 0), (3, 0), colors.Color(255 / 255, 182 / 255, 193 / 255, 0.5)),

    # cells (0,0) to (3,0) have hot pink font
    ('TEXTCOLOR', (0,0), (3, 0), colors.hotpink),

    # cells (0,1) to (1,4) have orange red font
    ('TEXTCOLOR', (0,1), (1,4), colors.orangered),

    # cells (2,1) to (3,4) have purple font
    ('TEXTCOLOR', (2,1), (3,4), colors.purple),

    # cells (1,3) to (2,4) have aqua font
    ('TEXTCOLOR', (1,3), (2,4), colors.aqua),

    # cells (0,0) to (3,0) have a left padding of 30px
    ('LEFTPADDING', (0,0), (3,0), 30),

    # cells (0,0) to (3,0) have a right padding of 30px
    ('RIGHTPADDING', (0, 0), (3, 0), 30),

    # cells (0,0) to (3,0) have a top padding of 30px
    ('TOPPADDING', (0, 0), (3, 0), 30),

    # cells (0,0) to (3,0) have a bottom padding of 30px
    ('BOTTOMPADDING', (0, 0), (3, 0), 30),

    # cells (3,1) to (3,4) are aligned right
    ('ALIGN', (3,1), (3,4), 'RIGHT'),

    # cells (0,0) to (3,0) use the "Helvetica", "Bold", "Italic" font, of size 18, with offset betwn lines of 15 
    ('FONT', (0, 0), (3,0), "Helvetica-BoldOblique", 18, 15),

    # cells (1,2) to (2,3) ause "Courier", "Italic" font, of size 12, with offset btwn lines of 15
    ('FONT', (1, 2), (2, 3), "Courier-Oblique", 12, 15),

    # cells (0,0) to (3,4) have grid lines, of width=2, and black
    ('GRID', (0, 0), (3, 4), 2, colors.black),

    # cells (2,2) to (3,4) have outer box lines, of width=6, in green
    ('BOX', (2, 2), (3, 4), 6, colors.green)
])

# TODO: 9. We will set the header style to the table.
pdf_table.setStyle(header_style)

# TODO: 10. We must build the table.
pdf.build([pdf_title, pdf_table])
