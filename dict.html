]

    {"bob"},

]

for i in range(len(dummy)):
    new_lst = []
    lst = list(dummy[i].items())

# Dicts to PDF
from reportlab.platypus import SimpleDocTemplate, TableStyle
from reportlab.platypus import Paragraph, Spacer, Table, Image
from reportlab.lib.styles import getSampleStyleSheet
from reportlab.lib import colors

dummy = [
    {'a': 3, 'b': 8, 'c': 4},
    {'a': 9, 'b': 4, 'c': 2},
    {'a': 9, 'b': 10, 'c': 6}
]
path = "../Resources/"
pdf = SimpleDocTemplate(path + "dummy_dicts.pdf")
styles = getSampleStyleSheet()

pdf_title = Paragraph("Dummy", styles["h1"])

dummy_dict = []
headers = ("k", "v")
dummy_dict.append(headers)

for i in range(len(dummy)):
    # print((dummy[i]))
    for k, v in dummy[i].items():
        # print((k, v))
        dummy_dict.append((k, v))
# print(dummy_list)
pdf_table = Table(data=dummy_dict)

header_style = TableStyle([
    ('ALIGN', (0, 0), (3, 0), 'CENTER'),
    ('FONT', (0, 0), (3, 0), "Helvetica-Bold", 16),
    ('GRID', (0, 1), (3, 9), 2, colors.black)
])

pdf_table.setStyle(header_style)

pdf.build([pdf_title, pdf_table])

    num SERIAL,
    keys VARCHAR(225),
    values VARCHAR(225)
    );""")

    {'a': 3, 'b': 8, 'c': 4},
    {'a': 9, 'b': 4, 'c': 2},
    {'a': 9, 'b': 10, 'c': 6}
]

for i in range(len(data)):
    for k, v in data[i].items():
        entries.append((k, v))

    (%s, %s)""", entries[i])
