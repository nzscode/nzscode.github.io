import csv
path = "../Resources/"

data = [["a", 1], ("b", "f"), [3, "c"], [4, 5], (9, 8)]
with open(path+"demo.csv", 'w', newline='') as file:
    field_names = ['a', 'b']
    dict_writer = csv.DictWriter(file, fieldnames=field_names)

    dict_writer.writeheader()
    
    for i in range(len(data)):
        dict_writer.writerow({'a': data[i][0], 'b': data[i][1]})

import json

data_random = "b", "c", "e", [5, 4], {"j": [1,2,3]}

    "x": 99,
    100: "y",
    "z": {
        "w": "6"
    }

    json.dump(data_random, file, indent=4)

with open(path + "demo_dict.json", 'w') as file:

    Important Info:

  • Dumps :
  • Converts object into a string
  • Can save as a string object, then write to a file.
  • Can write directly into a file.
  • Can print it directly to the console/terminal.
  • Considers lists/dicts/tuples/strings the same way.



    •                     
import json

data_random = "b", "c", "e", [5, 4], {"j": [1, 2, 3]}
data_list = ["a", "1", {"b": [3, "c"]}, ["d", "4"], ("e", 5)]
data_dict = {
    "x": 99,
    100: "y",
    "z": {
        "w": "6"
    }
}

path = "../Resources/"

# dumps converts the object into a string then saves directly to a file.
with open(path + "text_dump.json", 'w') as file:
    json_string = json.dumps(data_dict, indent=4)
    file.write(json_string)

# OR
with open(path + "text_dump.json", 'w') as file:
    file.write(json.dumps(data_dict, indent=4))

# Allows direct printing on console/terminal
print(json.dumps(data_dict, indent=4))
